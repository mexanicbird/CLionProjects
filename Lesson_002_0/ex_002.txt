#include <stdio.h>
#include <locale.h>

int main() {
    setlocale(LC_ALL, ""); // Подключаем ввод и вывод на русском языке
    // Первое задание
    int a;
    printf("Введите целое число: ");
    scanf("%d", &a);
    if(a >= 0 && a <= 100) {
        printf("Введенное число входит в диапазон от 0 до 100\n");
    } else {
        printf("Введенное число не входит в диапазон от 0 до 100\n");
    }


    // Второе задание
    int i = 0; // Счётчик чисел введенных пользователем
    int sum = 0; // Сумма чисел введённых пользователем
    double mean; // Среднее арифметическое
    printf("Введите 10 целых чисел для вычисления среднего арифметического: ");
    while(i < 10) {
        int a;
        scanf("%d", &a);
        sum = sum + a;
        i++;
    }
    mean = (double)sum / i;
    printf("Среднее арифметическое введенных чисел: %.1lf\n", mean);


    // Третье задание
    int b; // Число, введённое пользователем
    do {
        printf("Введите число не превышающее 1000: \n");
        scanf("%d", &b);
    } while(b <= 0 && b > 1000); // Проверка введённого числа на принадлежность диапазону от 0 до 1000
    for(int k = 1; k <= b; k++) { // Ввыод четных чисел
        if (k % 2 == 0) {
            printf("%d, ", k);
        }
    }
    printf("\n");


    // Четвертое задание
    int c; // Количество строк
    printf("Введите количество строк для вывода: ");
    scanf("%d", &c);
    for(int t = 0; t < c; t++) {
        int p = 0; // Количество символов в строке
        while(p <= (c * 2 - 1)) {
            if((p == (c - t)) || (p == c + t)) {
                printf("^");
            }  else {
                printf(" ");
            }
            p++;
        }
        if(t < c - 1) {
            printf("\n");
        }
    }
    return 0;
}
